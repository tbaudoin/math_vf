\begin{exercise}
    \begin{tasks}
        \task L'angle au sommet d'un triangle isocèle a
        une amplitude de \ang{50}, trouve l'amplitude des deux
        autres angles.
        \task Un des deux angles à la base d'un triangle isocèle a
        une amplitude de \ang{20}, trouve l'amplitude des deux
        autres.
        \task Un triangle rectangle a un angle dont l'amplitude
        vaut \ang{30}, trouve l'amplitude des deux autres angles.
        \task Un triangle isocèle a un angle droit, quelle est
        l'amplitude des deux autres ?
    \end{tasks}
\end{exercise}
