\newpage
Pour multiplier deux entiers : 
\begin{enumerate}[label=$\bullet$]
    \item \dotsforanswer{3}[Si les entiers sont de même signe,]
        \exemples
        \begin{itemize}[cols=2, itemsep=1cm, label=]
          \item $(-3)\cdot (-6)=$
          \item $7\cdot 6=$
          \item $-2\cdot (-3)=$
          \item $8\cdot 7=$
        \end{itemize}
        \vspace{1cm}
    \item \dotsforanswer{3}[Si les entiers sont de signes contraires,]
    \exemples
    \begin{itemize}[cols=2, itemsep=1cm, label=]
      \item $(-3)\cdot 6=$
      \item $-7\cdot 6=$
      \item $3\cdot -5=$
      \item $-10\cdot 3=$
    \end{itemize}
    \vspace{1cm}
\end{enumerate}
