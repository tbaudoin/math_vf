Pour multiplier deux entiers :
\begin{enumerate}[label=$\bullet$]
    \item \dotsforanswer{3}[Si les entiers sont de même signe,]
          \exemples
          \begin{itemize}[cols=2, itemsep=1cm, label=]
              \item $(-3)\cdot (-6)=$
              \item $7\cdot 6=$
              \item $-2\cdot (-3)=$
              \item $8\cdot 7=$
          \end{itemize}

              \item $-7\cdot 6=$

\end{enumerate}

\section{Exercices mélangés}

\begin{exercise}
    Vrai ou faux~? Justifie dans tous les cas.
    \begin{enumerate}
        \item \(( - 4) \cdot a\) est toujours négatif.
        \item \(a^{2}\) est toujours positif.
        \item Le produit de \(a\) par son opposé est négatif.
        \item Le double de \(a\) est positif.
    \end{enumerate}
\end{exercise}

\begin{exercise}
    Réponds aux questions suivantes.
    \begin{enumerate}
        \item Cicéron est né en l'an \(- 23\) et est mort en l'an \(38\). Combien de temps a-t-il vécu~?
        \item Antoine est né en l'an \(- 35\) et est mort à l'âge de \(57\) ans. En quelle année est-il mort~?
        \item L'empire de Césarius a été créé en \(- 330\) et s'est terminé en \(213\). Combien de temps a-t-il duré~?
        \item Antoniosus est mort en \(- 158\) à l'âge de \(63\) ans. En quelle année est-il né~?
    \end{enumerate}
\end{exercise}

\begin{exercise}
    Aux États-Unis, la température est mesurée en degrés Fahrenheit (\SI{}{\degree F}). Il est possible de convertir de les convertir en degré Celsius (\SI{}{\degreeCelsius}) à l'aide de la formule suivante,
    \[\SI{}{\degreeCelsius}=\frac{(\SI{}{\degree F} - 32)\cdot 5}{9}\]
    \begin{enumerate}
        \item A New-York, la température annoncée est de \SI{68}{\degree F}. Convertis  cette température en degrés Celsius à l'aide de la formule.

        \item Au même moment, la ville de Chicago relève une température de \SI{23}{\degree F}.Convertis-la aussi.

        \item  Bruxelles a mesuré une température de \SI{5}{\degreeCelsius} et au même moment, Flint, dans le Michigan, enregistrait une température de \SI{41}{\degree F}. Compare les deux températures.
    \end{enumerate}
\end{exercise}
