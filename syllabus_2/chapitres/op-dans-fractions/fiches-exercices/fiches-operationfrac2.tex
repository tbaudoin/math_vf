  \begin{exercise}
    Effectue.
    \begin{enumerate}[cols=3]
      \item $\left(\dfrac{5}{a} \cdot \dfrac{3}{4}\right) + \dfrac{1}{b}$, où $a = 2$ et $b = 3$
      \item $\dfrac{2}{x} - \left(\dfrac{7}{8} + \dfrac{1}{y}\right)$, où $x = 4$ et $y = 5$
      \item $\dfrac{3}{2} \cdot \left(\dfrac{4}{9} - \dfrac{2}{c}\right)$, où $c = 6$
      \item $\dfrac{5}{4} + \dfrac{3}{2} \cdot \left(\dfrac{2}{3} - \dfrac{1}{d}\right)$, où $d = 7$
      \item $\dfrac{3}{8} \cdot \dfrac{2}{5} - \left(\dfrac{1}{e} + \dfrac{3}{4}\right)$, où $e = 8$
      \item $\left(\dfrac{7}{9} + \dfrac{2}{3}\right)^2 - \dfrac{5}{6}$, où $f = 9$
      \item $\left(\dfrac{1}{2} \cdot \dfrac{3}{4}\right) + \dfrac{2}{g} - \dfrac{1}{3}$, où $g = 10$
      \item $\dfrac{4}{7} - \dfrac{1}{2} \cdot \left(\dfrac{5}{h} + \dfrac{2}{3}\right)$, où $h = 11$
      \item $\left(\dfrac{3}{5} - \dfrac{1}{4}\right) \cdot \dfrac{2}{i}$, où $i = 12$
      \item $\dfrac{2}{j} + \left(\dfrac{4}{5} \cdot \dfrac{1}{2}\right) - \dfrac{3}{8}$, où $j = 13$
      \item $\dfrac{5}{6} \cdot \left(\dfrac{7}{8} - \dfrac{2}{5}\right) + \dfrac{1}{4}$, où $k = 14$
      \item $\left(\dfrac{2}{3} - \dfrac{1}{5}\right) \cdot \dfrac{3}{4} + \dfrac{1}{2}$, où $l = 15$
      \item $\dfrac{3}{2} - \dfrac{5}{6} \cdot \left(\dfrac{1}{3} + \dfrac{2}{m}\right)$, où $m = 16$
      \item $\dfrac{1}{4} + \dfrac{2}{3} - \left(\dfrac{5}{6} \cdot \dfrac{3}{4}\right)$, où $n = 17$
      \item $\dfrac{2}{o} + \dfrac{1}{3} \cdot \left(\dfrac{4}{7} - \dfrac{3}{5}\right)$, où $o = 18$
      \item $\left(\dfrac{7}{9} + \dfrac{2}{3}\right) \cdot \left(\dfrac{1}{2} - \dfrac{3}{4}\right)$, où $p = 19$
      \item $\dfrac{3}{8} \cdot \left(\dfrac{2}{5} - \dfrac{1}{3}\right) + \dfrac{1}{4}$, où $q = 20$
      \item $\left(\dfrac{4}{7} - \dfrac{1}{2}\right) \cdot \dfrac{5}{6}$, où $r = 21$
      \item $\dfrac{3}{5} + \dfrac{1}{4} \cdot \left(\dfrac{2}{3} - \dfrac{3}{8}\right)$, où $s = 22$
      \item $\dfrac{4}{5} \cdot \left(\dfrac{1}{2} - \dfrac{3}{4}\right) + \dfrac{7}{9}$, où $t = 23$
    \end{enumerate}
  \end{exercise}
