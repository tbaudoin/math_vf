\begin{exercise}
	Coche la ou les bonne(s) réponse(s).\par
	\begin{enumerate}[itemsep=1.5em]

			      \choice $2a^2$
			      \choice $4a^2$
		      \end{qcm}

		\item Le volume d'un cube d'arête $a$ est
		      \begin{qcm}
			      \choice $3a$
			      \choice $a^2 \cdot a$
			      \choice $a^3$
		      \end{qcm}

		\item $20^0$ vaut
		      \begin{qcm}
			      \choice $1$
			      \choice $20$

		      \end{qcm}

		      \begin{qcm}

		      \end{qcm}

		      \begin{qcm}
			      \choice négatif
			      \choice positif
			      \choice sans avis
		      \end{qcm}
	\end{enumerate}
\end{exercise}
