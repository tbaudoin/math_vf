\begin{connaitre}[points=4]
    Les affirmations suivantes sont-elles vraies ou fausses? Justifie dans les deux cas.
    \begin{tasks}
        \task \(-3x > 2 \Leftrightarrow x > -\dfrac{2}{3}\).
        \task Le point $(0,8)$ appartient au graphe de la fonction $f(x)=2x + 8$.
        \task Si un système est indéterminé alors les deux droites qu'il représente sont parallèles.
        \task Si une fonction du premier degré est croissante, sa racine est toujours positive.
    \end{tasks}
\end{connaitre}
% SOLUTION %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{solconnaitre}
    \begin{tasks}
        \task Faux. Les deux membres ont été divisés par \(-3\), il faut donc changer le sens de l'inégalité.
        \task Vrai. Si \(x=0\) alors \(f(0) = 2\cdot 0 + 8 = 8\).
        \task Faux. Les deux droites sont confondues.
        \task Faux. Cela dépend du signe de son ordonnée à l'origine, si elle est positive alors la racine est négative sinon la racine est positive.
    \end{tasks}
\end{solconnaitre}

\begin{appliquer}[points=6]
    Détermine l'expression analytique des fonctions suivantes grâce aux informations données. N'oublie pas d'écrire l'expression complète de la fonction.
    \begin{tasks}
        \task Le graphe de la fonction est une droite qui passe par les points \((3,1)\) et \((-3,-5)\).
        \task La fonction admet \(-2\) comme racine et son graphe passe par le point \((2,10)\).
        \task Un tableau de valeur de la fonction est le suivant :
        \begin{center}
            \begin{tblr}{hlines, vline{1,2,5}, columns={mode=math, c, 1cm}}
                x    & -2 & 4 & 10 \\
                f(x) & 5  & 8 & 11 \\
            \end{tblr}
        \end{center}
    \end{tasks}
\end{appliquer}
% SOLUTION %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{solappliquer}
    % 1 point pour $m$ et 1 point pour $p$.\\
    % -0,5 point si l'expression entière n'est pas notée.
    \begin{tasks}
        \task
        \begin{tblr}[t]{columns={leftsep=0pt},
                row{1}={font=\bfseries},
                colspec={XX},
                cell{3}{1}={r=1,c=2}{c}
            }
            Déterminons m                        & Déterminons p         \\
            $m = \dfrac{12}{-8} = -\dfrac{3}{2}$ &
            $(-2;8)\in f_1 \Rightarrow \begin{aligned}[t]
                                               8 & = -\dfrac{3}{2}\cdot (-2) + p \\
                                               8 & = 3 + p                       \\
                                               p & = 5                           \\
                                           \end{aligned}$ \\
            \fbox{$f_1(x) = -\dfrac{3}{2}x + 5$} &
        \end{tblr}
        \task
        \begin{tblr}[t]{columns={leftsep=0pt},
                row{1}={font=\bfseries},
                colspec={XX},
                cell{3}{1}={r=1,c=2}{c}
            }
            Déterminons m                      & Déterminons p \\
            $m = \dfrac{14}{4} = \dfrac{7}{2}$ &
            $f_2$ est linéaire $\Rightarrow p = 0$             \\
            \fbox{$f_2(x) = \dfrac{7}{2}x$}    &
        \end{tblr}
        \task
        \begin{tblr}[t]{columns={leftsep=0pt},
                row{1}={font=\bfseries},
                colspec={XX},
                cell{3}{1}={r=1,c=2}{c}
            }

            $G_{f_3} \parallel G_g \Rightarrow m = -5$ &

                                               p & = -10               \\
