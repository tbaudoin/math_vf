\begin{exercise}[points=5]
    Les propositions suivantes sont-elle vraies ou fausses? Justifie dans les deux cas.
    \begin{enumerate}
        \item $\dfrac{1}{\sqrt{2}}=\dfrac{\sqrt{2}}{2}$
        \item \input{tikz/triangle_rectangle_vrai_faux.tikz} $\Rightarrow c^2 = a^2+b^2$
        \item Deux nombres opposés ont la même racine carrée.
        \item $\sqrt{a} + \sqrt{b} = \sqrt{a+b}$
        \item $2\sqrt{3} + 4\sqrt{3} = \sqrt{108}$
    \end{enumerate}
\end{exercise}

\begin{solution}
    \begin{enumerate}
        \item Vrai, $\dfrac{1}{\sqrt{2}}\cdot\dfrac{\sqrt{2}}{\sqrt{2}} = \dfrac{\sqrt{2}}{2}$
        \item Faux, $c^2 = a^2 - b^2$
        \item Faux, un nombre négatif n'admet pas de racine carré dans $\mathbb{R}$.
        \item Faux, $\sqrt{4} + \sqrt{16} = 6$ et $6\ne \sqrt{4+16}=\sqrt{20}$
        \item Vrai. $2\sqrt{3} + 4\sqrt{3} = 6\sqrt{3} = \sqrt{3\cdot 36} = \sqrt{108}$
    \end{enumerate}
\end{solution}

% 2pts pour vérifier que l'angle est droit par pythagore, 1 pt pour pythagore dans BDE
\begin{exercise}[points=3]
    Sur le dessin ci-dessous, les points $A$, $B$ et $E$ sont alignés et $C$ est le milieu de $[BD]$. Calculer $\left| ED \right|$.
    \begin{center}
        \input{tikz/deux_triangles.tikz}
    \end{center}
\end{exercise}
\begin{solution}
    $|CB| = 3$ car $C$ est le milieu de $[DB]$ et que $|DC| = 3$.

    Par conséquent, le triangle $ABC$ est rectangle par la réciproque du théorème de Pythagore. En effet,
    \begin{align*}
        \sqrt{34}^2 &= 5^2 + 3^2\\
        34 &= 25 + 9
    \end{align*}

    Les point $ABE$ sont alignés et $|\widehat{ABC}| = 90\degres$. Donc, $|\widehat{EBD}| = 90\degres$. Par conséquent, le triangle $ABE$ est aussi rectangle.

    En appliquant le théorème de Pythagore dans ce triangle on trouve,
    $|DE| = \sqrt{85}$
    \newpage
\end{solution}

\begin{exercise}[points=3]
    Voici une photo aérienne du parc de Jamblinne. En sachant que la vitesse de marche moyenne est de 1,5 m/s, détermine le temps gagné (en seconde) par un piéton s'il coupe à travers le parc.
    \begin{center}
        \includegraphics[width=.5\linewidth]{img/dejamblinne.png}
    \end{center}
\end{exercise}
\begin{solution}
    Si le piéton contourne le parc il parcours : $28,45 + 30,87 = 59,23$~m. Il mettra donc $\dfrac{59,23}{1,5}=39,55$~s pour parourir cette distance.

    Si le piéton coupe par le parc il parcours : $\sqrt{28,45^2 + 30,87^2}=41,98$~m (Théorème de pythagore). Il mettra donc $\dfrac{41,98}{1,5} = 27,99$~s.

    Le piéton qui coupe par le parc gagne donc $39,55 - 27,99 = 11,56$~s.
\end{solution}

\begin{exercise}[points=3]
    Madame Regout vient à l'école en vélo. Elle voudrait qu'on installe une rampe d'accès sur les marches pour monter facilement son vélo sur le seuil de la porte d'entrée qui se situe à 40 cm du sol. Devant l'école un espace de 1,5 m est disponible pour installer cette rampe. Si la rampe mesure 1,5 m :
    \begin{itemize}
        \item Quelle sera l'amplitude de l'angle d'inclinaison entre la rampe et le sol ?
        \item Y aura-t-il assez d'espace devant l'école pour placer la rampe ?
    \end{itemize}
\end{exercise}
\begin{solution}
    \begin{itemize}
        \item 
        L'amplitude de l'angle d'inclinaison vaut $\arcsin\left(\dfrac{40}{150}\right)=15,47\degres$.
        \item 
        Place occupée par la rampe : $\sqrt{150^2 - 40^2} = 144,57$~m. La rampe occuper $144,57$~cm il y aura donc assez d'espace avec $150$~cm.
    \end{itemize}
\end{solution}
