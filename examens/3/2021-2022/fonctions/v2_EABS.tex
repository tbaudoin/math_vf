\begin{exercise}[points=4]
    Les affirmations suivantes sont-elles vraies ou fausses. Justifie dans les deux cas.
    \begin{enumerate}[itemsep=1em]
        \item Le point $(2;10)$ appartient au graphe de la fonction $f(x)=3x+4$.
        \item $f(x) = 2x$ est une fonction linéaire.
        \item L'ordonnée à l'origine d'une fonction est le point d'intersection de la fonction avec l'axe des ordonnées.
        \item $3$ est la racine de la fonction $f(x)=-2x-6$
    \end{enumerate}
    \vspace*{1em}
\end{exercise}
% SOLUTION %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{solution}
    1 point par justification correcte.
    \begin{enumerate}
        \item Vrai.
              $\begin{aligned}[t]
                      10 & \stackrel{?}{=} 3\cdot 2 + 4 \\
                      10 & \stackrel{?}{=} 6 + 4        \\
                      10 & \ne 10
                  \end{aligned}$
        \item Vrai. Le graphe de cette fonction passe par $(0;0)$ car $f(x) = 2\cdot 0 = 0$.
        \item Faux. L'ordonnée à l'origine d'une fonction est l'ordonnée du point d'intersection de la fonction avec l'axe des ordonnées.
        \item Faux. La racine est la valeur de $x$ qui annule $f(x)$. Or $-2\cdot 3 - 6 = 12 \ne 0$.
    \end{enumerate}
\end{solution}

\begin{exercise}[points=10]
    Détermine l'expression analytique des fonctions suivantes grâce aux informations données. Note ton raisonnement et n'oublie pas d'écrire l'expression complète de la fonction.
    \begin{enumerate}[itemsep=2em]
        \item
              $A(4;1)$ et $B(8;9)$ appartiennent à la fonction $f_1$.
        \item
              Voici un tableau de valeur de la fonction $f_2$
              \hfil
              $\begin{tblr}{hlines, vline{1,2,5},columns={c}}
                      x    & 1 & 8   & -2  \\
                      f(x) & 0 & 21 & 9
                  \end{tblr}$
        \item
              La racine de la fonction $f_3$ est $-5$ et son graphe est parallèle à celui de la fonction $g(x)=-2x + 10$.
        \item
              La fonction $f_4$ est linéaire et passe par $(2;8)$.
        \item
              Le graphe de la fonction $f_5$ est le suivant\hfil \input{tikz/m_p_fct01.tikz}
    \end{enumerate}
\end{exercise}
% SOLUTION %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{solution}
    1 point pour $m$ et 1 point pour $p$.\\
    -0,5 point si l'expression entière n'est pas notée.
    \raggedcolumns
    \begin{enumerate}
        \item $f_1(x) = 4x-7$

              \begin{multicols}{2}
                  \textbf{$\bm{m}$ ?}
                  \medbreak
                  $m=\dfrac{9-1}{8-4}=\dfrac{8}{2}=4$

                  \textbf{$\bm{p}$ ?}
                  \medbreak
                  $\begin{aligned}[t]
                          A(4;1)\in f_1 \Rightarrow 1 & = 2\cdot 4 + p \\
                          p                           & = -7
                      \end{aligned}$
              \end{multicols}
        \item $f_2(x) = -3x-3$
              \begin{multicols}{2}
                  \textbf{$\bm{m}$ ?}
                  \medbreak
                  $m=\dfrac{21-0}{8-1}=\dfrac{21}{7}=3$

                  \textbf{$\bm{p}$ ?}
                  \medbreak
                  $\begin{aligned}[t]
                          (1;0)\in f_2 \Rightarrow 0 & = 3\cdot 1 + p \\
                          p                             & = -3
                      \end{aligned}$
              \end{multicols}
        \item $f_3(x) = -2x - 10$
              \begin{multicols}{2}
                  \textbf{$\bm{m}$ ?}
                  \medbreak
                  $G_f \parallel G_g \Rightarrow m = -2$
                  \columnbreak

                  \textbf{$\bm{p}$ ?}
                  \medbreak
                  $\begin{aligned}[t]
                          \text{La racine est }-5\Rightarrow 0 & = -2\cdot (-5) + p \\
                          p                                    & = -10
                      \end{aligned}$
              \end{multicols}
        \item $f_4(x) = 4x$
              \begin{multicols}{2}
                  \textbf{$\bm{m}$ ?}
                  \medbreak
                  $f_4$ est linéaire $\Rightarrow (0;0) \in f_4$
                  \medbreak
                  $m = \dfrac{8-0}{2-0} = \dfrac{8}{2} = 4$
                  \columnbreak

                  \textbf{$\bm{p}$ ?}
                  \medbreak
                  $f_4$ est linéaire $\Rightarrow p=0$
              \end{multicols}
        \item $f_5 = -\dfrac{1}{4}x + 1 $
              \begin{multicols}{2}
                  \textbf{$\bm{m}$ ?}
                  \medbreak
                  $m=\dfrac{-1}{4}$
                  \columnbreak

                  \textbf{$\bm{p}$ ?}
                  \medbreak
                  $p=1$
              \end{multicols}
    \end{enumerate}
    \newpage
\end{solution}

\begin{exercise}[points=4]
    Le graphique ci-dessou représente l'évolution du prix de deux congélateurs (AEG et BOSCH) en fonction de leur coût énergétique annuel. 
    
    Un congélateur consomme de l'électricité, il y a donc un coût annuel à ajouter au prix d'achat pour le fonctionnement de l'appareil.
    
    À l'aide des ces informations, réponds aux questions suivantes en expliquant ton raisonnement.
    \begin{enumerate}[itemsep=1em]
        \item Quel est le prix initial des deux congélateurs ?
        \item Quelle est le coût énergétique annuel (en €/an) de chaque congélateur ?
        \item En estimant la durée de vie d', after=\vspace*{1em}]un congélateur à dix ans. Quel sera le congélateur le plus avantageux économiquement parlant ?
    \end{enumerate}
    \begin{center}
        \input{tikz/congelateurs.tikz}
    \end{center}
\end{exercise}
% SOLUTION %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{solution}
    \begin{enumerate}
        \item Le BOSCH coûte 700~€ et l'AEG 500~€. Il faut regarder le prix pour 0 années.
        \item Le BOSCH coûte 200~€ sur 4 ans. Il coûte donc 50~€/an.\\
        Le AEG coûte 300~€ sur 4 ans. Il coûte donc 75~€/an.
        \item Après 10 ans le BOSCH aura coûté $700 + 50\cdot 10 = 1200$~€ et le AEG $500 + 75\cdot 10 = 1250$~€. Le BOSCH est donc économiquement plus avantageux.
    \end{enumerate}
\end{solution}
