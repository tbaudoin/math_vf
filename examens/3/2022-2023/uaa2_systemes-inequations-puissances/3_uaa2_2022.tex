\section*{\fbox{C2} Appliquer une procédure}
\begin{appliquer}
    Résous le système suivant par substitution.
    \begin{center}
        \systeme{2x + 5y &= 10}{x-4y&=-8}
    \end{center}
\end{appliquer}
%
\begin{solappliquer}
    \systeme{2x + 5y &= 10}{x-4y&=-8}

    Isolons $x$ dans la deuxième équation,

    \systeme{2x + 5y &= 10}{x&=-8+4y}

    Remplaçons $x$ dans la première équation par son expression dans la seconde,

    \systeme{2(-8+4y) + 5y &= 10}{x&=-8+4y}

    Résolvons la première équation,

    \begin{equation*}
        \begin{aligned}
            2(-8+4y) + 5y & = 10 \\
            -16 + 8y + 5y & = 10 \\
            -16 + 13 y    & = 10 \\
            13 y          & = 26 \\
            y             & = 2
        \end{aligned}
    \end{equation*}

    Maintenant que nous connaisson la valeur de $y$, nous pouvons trouver celle de $x$ en remplaçant $y$ par sa valeur dans la deuxième équations,

    \begin{equation*}
        \begin{aligned}
            x & = -8 + 4 \cdot 2 \\
            x & = -8 + 8         \\
            x & = 0
        \end{aligned}
    \end{equation*}
\end{solappliquer}

\begin{appliquer}
    Résous les inéquations suivantes en notant les solutions sous forme d'intervalles.
\end{appliquer}

\begin{appliquer}
    Dans chaque cas, détermine l'intersection des deux intervalles de réels
\end{appliquer}

\end{appliquer}
