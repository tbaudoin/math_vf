\begin{connaitre}[points=5]
    Vrai ou faux ? Justifie dans les deux cas.
    \begin{tasks}
        \task La moitié de \(2^{-6}\) est \(2^{-3}\) ?
        \task $12,35\cdot 10^2$ est l'écriture scientifique de \num{1235}
        \task Si deux droites sont parallèles alors le système d'équations qui les représentent admet une solution unique.
        \task Si $-2x > 20$ alors $x > -10$.
        \task $x<8$ admet comme solutions $S=]-\infty; 8]$.
    \end{tasks}
\end{connaitre}
\begin{solconnaitre}
    \begin{tasks}
        \task Faux. $\dfrac{2^{-6}}{2} = 2^{-7}$.
        \task Faux. $1, 235 \cdot 10^2$ qui est l'écriture décimale de \num{1235}.
        \task Faux. Un tel système n'a pas de solution.
        \task Faux. $x<-10$.
        \task Faux. Les solutions sont $S=]-\infty; 8[$
    \end{tasks}
\end{solconnaitre}
