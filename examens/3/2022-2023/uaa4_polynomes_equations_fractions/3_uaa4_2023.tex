\section*{\fbox{C1} Expliciter les savoirs et les procédures}
\begin{connaitre}
    Soient deux polynômes,
    \(A(x) = 4x^4 + 2x^2 - 3\) et \(B(x) = 3x^3 - x^2 + 2x + 1\).
    \begin{enumerate}
        \item Quel est le degré de \(A(x)\) ?
        \item Que vaut le terme indépendant de \(B(x)\) ?
        \item Quels sont les coefficients de \(A(x)\) ?
        \item Sans effectuer, que vaut le degré de \(A(x) + B(x)\) ?
        \item Sans effectuer, que vaut le degré de \(A(x) \cdot B(x)\) ?
        \item Sans effectuer, que vaut le degré de \(\dfrac{A(x)}{B(x)}\) ?
        \item Détermine la valeur numérique de \(A(x)\) si \(x=-1\).
    \end{enumerate}
\end{connaitre}

\begin{connaitre}
    Que doit valoir \(a\) pour que \(x^3 - x^2 - 9x + a\) soit divisible par \(x-3\) ? Justifiez.
\end{connaitre}

\begin{connaitre}
    Donnez un exemple d'équation du second degré qui admet uniquement \(-1\) et \(2\) comme solution.
\end{connaitre}

\begin{connaitre}
    Donnez un exemple d'une fraction algébrique dont les conditions d'existence sont \(x\ne 0\), \(x\ne 3\) et \(x\ne -3\).
\end{connaitre}

\section*{\fbox{C2} Appliquer une procédure}
\begin{appliquer}
    Résolvez les équations suivantes.
    \begin{enumerate}
        \item \(x^2=64\) % (x-8)(x+8)
        \item \(x^2 + 6x + 9 = 0\) % (x+3)^2
        \item \(x^3 + 2 x^2 + x = 0\) % x(x+1)^2
        \item \(x^2 -3x - 28 = 0\) % (x+4)(x-7)
        \item \(x³ - 4x² - 11x + 30 = 0\) %(x-2)(x+3)(x-5)
    \end{enumerate}
\end{appliquer}

\begin{appliquer}
    Donnez les conditions d'existence puis effectuez l'opération demandée. N'oubliez pas de simplifier votre réponse finale.
    \begin{enumerate}
        \item \(\dfrac{4x+12}{2x^2 - 2} \cdot \dfrac{8}{x^2-9}\)
    \end{enumerate}
\end{appliquer}
