\section*{\fbox{C1} Expliciter les savoirs et les procédures}
\begin{connaitre}[points=3]
    Soient deux polynômes,
    \(A(x) = 4x^4 + 2x^2 - 3\) et \(B(x) = 3x^3 - x^2 + 2x + 1\).
    \begin{enumerate}
        \item Quel est le degré de \(A(x)\) ?
        \item Que vaut le terme indépendant de \(B(x)\) ?
        \item Quels sont les coefficients de \(A(x)\) ?
        \item Sans effectuer, que vaut le degré de \(A(x) + B(x)\) ?
        \item Sans effectuer, que vaut le degré de \(A(x) \cdot B(x)\) ?
        \item Lequel de ces deux polynômes est complet ?
        % \item Détermine la valeur numérique de \(A(x)\) si \(x=-1\).
    \end{enumerate}
\end{connaitre}

\begin{connaitre}[points=1]
    Donnez un exemple d'équation du second degré qui n'admet pas de solution.
\end{connaitre}

\begin{connaitre}[points=2]
    Parmi les expressions suivantes, quelles sont celles qui sont écrites sous leur forme factorisée ?
    \begin{multicols}{3}
        \begin{enumerate}
            \item \(5x-2\)
            \item \((x-1)(2x-3)\)
            \item \((5x-2)^2\)
            \item \(3xy+2x\)
            \item \(\left(\dfrac{3x}{2}-2y\right)y\)
            \item \((5x-2)^2+x\)
            \item \(\dfrac{3x}{2}-2xy\)
            \item \(x^2yz\)
            \item \(3x^2+2x-5\)
        \end{enumerate}
    \end{multicols}
\end{connaitre}

\section*{\fbox{C2} Appliquer une procédure}
\begin{appliquer}[points=11]
    Résolvez les équations suivantes.
    \begin{multicols}{2}
        \begin{enumerate}
            \item \(x^2 = 64\) % (x-8)(x+8) 1 pt
            \item \(x^2 + 6x + 9 = 0\) % (x+3)^2 1pt
            \item \(x^3 + 2 x^2 + x = 0\) % x(x+1)^2 2 pts
            \item \(x^2 -3x = 28 \) % (x+4)(x-7) 2 pts
            \item \(2x^2-3x-20=0\) % (2x+5)(x-4) 2pts
            \item \(x^3 - 4x^2 - 11x + 30 = 0\) %(x-2)(x+3)(x-5) 3 pts
        \end{enumerate}
    \end{multicols}
\end{appliquer}
