\begin{exercise}[points=4]
    Coche la ou les bonnes réponses. Si tu oublies des bonnes réponses ou que tu coches des mauvaises réponses, tu n'obtiens aucun point pour l'énoncé.
    \begin{enumerate}
        \item Une fonction linéaire est une fonction qui \dots
              \begin{QCM}
                  \choice passe par l'origine du repère
                  \choice a une pente nulle
                  \choice est toujours croissante
                  \choice admet $0$ comme racine.
              \end{QCM}
        \item Combien de solution(s) admet l'équation $x^2+4=0$ ?
              \begin{QCM}(3)
                  \choice $2$
                  \choice $1$
                  \choice $0$
              \end{QCM}
        \item Quel est le degré du polynôme $x^6 + 5x^4 -2x^3 +23$ ?
              \begin{QCM}(3)
                  \choice $3$
                  \choice $4$
                  \choice $6$
              \end{QCM}
        \item Quelle(s) expression(s) sont équivalentes à $\dfrac{3}{\sqrt{2}}$ ?
              \begin{QCM}(4)
                  \choice $\dfrac{3\sqrt{2}}{2}$
                  \choice $\dfrac{\sqrt{6}}{2}$
                  \choice $\sqrt{\dfrac{3}{2}}$
                  \choice $\sqrt{\dfrac{9}{2}}$
              \end{QCM}
    \end{enumerate}
\end{exercise}
% SOLUTION
\begin{solution}
    Évaluation :
    \begin{itemize}
        \item Les bonnes réponses sont cochées et aucune mauvaise réponse n'est cochée \hfill 1
    \end{itemize}
    \begin{enumerate}
        \item Une fonction linéaire est une fonction qui \dots
              \begin{QCM}
                  \choice[\faCheckSquare] passe par l'origine du repère
                  \choice a une pente nulle
                  \choice est toujours croissante
                  \choice[\faCheckSquare] admet $0$ comme racine.
              \end{QCM}
        \item Combien de solution(s) admet l'équation $x^2+4=0$ ?
              \begin{QCM}(3)
                  \choice $2$
                  \choice $1$
                  \choice[\faCheckSquare] $0$
              \end{QCM}
        \item Quel est le degré du polynôme $x^6 + 5x^4 -2x^3 +23$ ?
              \begin{QCM}(3)
                  \choice $3$
                  \choice $4$
                  \choice[\faCheckSquare] $6$
              \end{QCM}
        \item Quelle(s) expression(s) est/sont équivalentes à $\dfrac{3}{\sqrt{2}}$ ?
              \begin{QCM}(4)
                  \choice[\faCheckSquare] $\dfrac{3\sqrt{2}}{2}$
                  \choice $\dfrac{\sqrt{6}}{2}$
                  \choice $\sqrt{\dfrac{3}{2}}$
                  \choice[\faCheckSquare] $\sqrt{\dfrac{9}{2}}$
              \end{QCM}
    \end{enumerate}
\end{solution}

\begin{exercise}[points=2]
    Après avoir lu la légende, complète le tableau.

    Type de fonction \hfill A : affine \hfill L : linéaire \hfill C : constante

    Croissance de la fonction \hfill $\nearrow$  : croissante \hfill $\searrow$ : décroissante \hfill C : constante

    \begin{tblr}{hlines,
            vlines,
            columns = {2cm,c},
            rows={.9cm,m},
            column{1} = {3cm, l},
            cell{1}{1} = {c},
            rowsep=1ex}
        Expression analytique & Type de fonction & Pente & Croissance & Racine & Ordonnée à l'origine \\
        % $f(x) = 2x-2$                                                                                 \\
        % $f(x) = 5$                                                                                    \\
        $f(x) = 3x$                                                                                   \\
        %$f(x) = 8 - 3x$                                                                               \\
        $f(x) = \dfrac{2x + 8}{4}$                                                                    \\
    \end{tblr}
\end{exercise}

\begin{solution}
    Évaluation : 
    \begin{itemize}
        \item Il n'y pas d'erreur dans a ligne \hfill 1
        \item Il y a une erreur dans la ligne \hfill 0,5
    \end{itemize}
    \begin{tblr}{hlines,
            vlines,
            columns = {2cm,c},
            rows={.9cm,m},
            column{1} = {3cm, l},
            cell{1}{1} = {c},
            rowsep=1ex}

        $f(x) = 3x$                & L                & 3              & $\nearrow$ & 0      & 0                    \\
        % $f(x) = 8 - 3x$            & A                & $-3$           & $\searrow$ & $\dfrac{8}{3}$ & 8                    \\
        $f(x) = \dfrac{2x + 8}{4}$ & A                & $\dfrac{1}{2}$ & $\nearrow$ & $-4$   & 2                    \\
    \end{tblr}
\end{solution}
