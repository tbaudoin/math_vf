	\section{Calcul mental}
		\subsection{Distributivité simple}	
			\begin{bclogo}[arrondi=0.3, logo=\bccle]{Définition}
				Pour multiplier une somme/une différence, il faut multiplier chaque terme de la somme par ce nombre et additionner/soustraire les résultats obtenus.\\
				Exemples : 
				\parbox[t]{0.3\textwidth}{
					\begin{align*}
						6\cdot (10+7) &= 6\cdot 10 + 6\cdot 7\\
						& = 60 + 42\\
						& = 102
					\end{align*}}
				\parbox[t]{0.3\textwidth}{
					\begin{align*}
						6\cdot (10-7) &= 6\cdot 10 - 6\cdot 7\\
						& = 60 - 42\\
						& = 18
					\end{align*}}\\
			\end{bclogo}

	\subsection{Priorités des opérations}
		\begin{bclogo}[arrondi=0.3, logo=\bccle]{Définition}
			Dans un calcul, on effectue en priorité :
			\begin{enumerate}
				\item les calculs entre parenthèses (\textbf{P} pour parenthèse)
				\item les puissances (\textbf{E} pour exposant)
				\item de gauche à droite, les produits et les quotients (\textbf{M} pour multiplication et \textbf{D} pour division)
				\item de gauche à droite, les sommes et les différences (\textbf{A} pour addition et \textbf{S} pour soustraction)
			\end{enumerate}
			Moyen mnémotechnique : \textbf{PEMDAS}	
		\end{bclogo}

\section{Les nombres entiers}
	 \subsection{Opérations sur les entiers}
		\begin{bclogo}[arrondi=0.3,logo=\bccle]{Opérations sur les entiers}
			\begin{enumerate}[label=\arabic*)]
				\item Somme de deux nombres entiers :
				\begin{enumerate}
					\item Pour additionner deux nombres entiers de \textbf{même signe}, on \textbf{conserve} le signe et on \textbf{additionne} les valeurs absolues.\\
					Exemples : 
					\parbox{0.3\linewidth}{$(+2)+(+3)=+5$}
					\parbox{0.3\linewidth}{$(-2)+(-5)=-7$}
					\item Pour additionner deux nombres entiers de \textbf{signes différents}, on \textbf{donne} à la somme le \textbf{signe du terme} ayant la plus \textbf{grande valeur absolue} et on \textbf{soustrait} les valeurs absolues (la plus grande moins la plus petite)\\
					Exemples :
					\parbox{0.3\linewidth}{$(-6)+(+2)=-4$}
					\parbox{0.3\linewidth}{$(-2)+(+5)=+3$}
				\end{enumerate}
				\item Règle des signes successifs :
				\begin{enumerate}
					\item On remplace deux signes successifs \textbf{identiques} par un signe \textbf{$+$}.
					\item On remplace deux signes successifs \textbf{opposés} par un signe \textbf{$-$}
				\end{enumerate}
			\end{enumerate}
		\end{bclogo}
