\begin{exercise}

    Un stade de football propose deux formules tarifaires. Combien de fois
    faut-il aller au stade pour que la seconde formule soit aussi intéressante
    que la première ?
    \begin{center}
        \begin{tabularx}{.9\linewidth}{YY}
            \toprule
            \textbf{Formule 1} & \textbf{Formule 2}               \\
            \midrule
            Entrée : 20 €      & Carte de membre annuelle : 100 € \\
                               & Entrée : 15 €                    \\
            \bottomrule
        \end{tabularx}
    \end{center}
\end{exercise}
\begin{solution}
    \begin{itemize}[label=\faAngleRight]
        \item $x$ : le nombre d'entrée au stade
        \item $F_1$ : le prix payé avec la formule 1, $20x$
        \item $F_2$ : le prix payé avec la formule 2, $100 + 15x$
    \end{itemize}
    Que doit valoir $x$ pour que $F_1=F_2$ ?
    \begin{align*}
        F_1 & = F2      \\
        20x & = 100+15x \\
        5x  & = 100     \\
        x   & = 20
    \end{align*}
    Pour que les deux formules soient aussi intéressantes l'une que l'autre,
    il faut aller 20 fois au stade.
\end{solution}

\begin{exercise}
    Un groupe de 60 élèves de deuxième année part au théâtre avec 4 enseignants
    et 2 éducateurs. Une place adulte est 6 euros plus chère qu’une place
    étudiant. L’école a payé 564 euros au total.\par
    Quel est le prix d’une place étudiant et d’une place adulte ?
\end{exercise}
\begin{solution}
    \begin{itemize}[label=\faAngleRight]
        \item $x$ : le prix d'une place étudiante
        \item $x+6$ : le prix d'une place adulte
        \item $E$ : le prix pour les élèves, $60x$
        \item $A$ : le prix pour les adultes, $6(x+6)$
    \end{itemize}
    \begin{align*}
        60x + 6(x+6)  & = 564 \\
        60x + 6x + 36 & = 564 \\
        66x + 36      & = 564 \\
        66x           & = 528 \\
        x             & = 8
    \end{align*}
    Une place étudiant coûte 8€ et une place adulte coûte 14€.
\end{solution}

\begin{exercise}

    pour 3 de ses clients.\\
    Le 3\ieme client à demandé 300 pommes de plus que le 2\ieme.\\
    Le 2\ieme en a demandé 200 de plus que le 1\ier.\par

\end{exercise}
\begin{solution}
    \begin{itemize}[label=\faAngleRight]
        \item $x$ : le nombre de pomme donné au premier client
        \item $x+200$ : le nombre de pomme donné au deuxième client
        \item $x+500$ : le nombre de pomme donné au troisième client

    \begin{align*}
        x+(x+200)+(x+500) & =\np{1600}  \\
        3x + 700          & = \np{1600} \\

    \end{align*}
    Le marchand doit donnée 300 pommes au premier client.
\end{solution}

\begin{exercise}
    Ikram a consommé les $\dfrac{3}{4}$ de son abonnement gsm
    la première quinzaine du mois de juin.
    Elle consomme les $\dfrac{2}{5}$ de ce qu'il reste du forfait
    la seconde quinzaine.
    \begin{tasks}
        \task Calcule la part de son abonnement qu'elle a consommée durant tout
        son mois.
        \task Calcule la part de son abonnement qu'elle n'a pas consommée.
        \task Il lui reste 12 minutes à la fin du mois. Quel est le nombre de
        minutes disponibles au début du mois de juin ?
    \end{tasks}
\end{exercise}
\begin{solution}
    \begin{tasks}
        \task La première quinzaine elle a consommé $\dfrac{3}{4}$ de son
        abonemment, il lui reste donc $\dfrac{1}{4}$. La deuxième quinzaine,
        elle a consommé $\dfrac{2}{5}$ de ce qu'il lui restait, c'est-à-dire,
        \begin{equation*}
            \dfrac{2}{5}\cdot\dfrac{1}{4} = \dfrac{1}{10}
        \end{equation*}
        Elle aura donc consommée durant tout le mois,
        \begin{equation*}
            \dfrac{3}{4} + \dfrac{1}{10} = \dfrac{17}{20}
        \end{equation*}

        \task Il lui reste $\dfrac{3}{20}$ non consommé.
        \task En utilisant la règle de trois,
        \begin{align*}
            \dfrac{3}{20}  & \longrightarrow \SI{12}{min} \\
            \dfrac{1}{20}  & \longrightarrow \SI{4}{min}  \\
            \dfrac{20}{20} & \longrightarrow \SI{80}{min}
        \end{align*}
        Au début du mois de juin \SI{80}{min} sont disponibles.
    \end{tasks}

\begin{exercise}
    Détermine l'amplitude de l'angle demandé en justifiant ta réponse.
    \begin{tasks}(2)
        \task $\widehat{A_1}=40\degres$, $\widehat{A_2} = ?$.\\\inputtikz{angle_1}
        \task $\widehat{A_1}=60\degres$, $\widehat{B_1} = ?$.
        \\$d_1\sslash d_2$\\\inputtikz{angle_2}
            \task $\widehat{A_1}=45\degres$, $\widehat{B_2} = ?$, $\widehat{C_1}=?$.
            \\$ABCD$ est un parallélogramme\\\inputtikz{angle_3}
    \end{tasks}
\end{exercise}
\begin{solution}
    \begin{tasks}
        \task
        $\widehat{A_1}$ et $\widehat{A_2}$ forment un angle plat, donc
        \begin{align*}
            \widehat{A_2} & = 180\degres - \widehat{A_1} \\
            \widehat{A_2} & = 180\degres - 40\degres     \\
            \widehat{A_2} & = 140\degres                 \\
        \end{align*}
        \task
        Soit $\widehat{B_2}$ l'angle correspondant à $\widehat{A_1}$.
        $d_1\sslash d_2$ donc,
        \begin{equation*}
            \widehat{B_2} = \widehat{A_1} = 60\degres
        \end{equation*}
        $\widehat{B_2}$ et $\widehat{B_1}$ forment un angle plat, donc
        \begin{align*}
            \widehat{B_1} & = 180\degres - \widehat{B_2} \\
            \widehat{B_1} & = 180\degres - 60\degres     \\
            \widehat{B_1} & = 120\degres                 \\
        \end{align*}
        \task
        $\widehat{A_1}$ et $\widehat{B_2}$ sont correspondants et
        $AD\sslash BC$ (propriétés d'un parallélogramme), donc
        \begin{equation*}
            \widehat{B_2}=\widehat{A_1}=45\degres
        \end{equation*}
        $\widehat{B_1}$ et $\widehat{B_2}$ forment un angle plat, donc
        \begin{align*}
            \widehat{B_1} & = 180\degres - \widehat{B_2} \\
            \widehat{B_1} & = 180\degres - 45\degres     \\
            \widehat{B_1} & = 135\degres                 \\
        \end{align*}
        $\widehat{B_1}$ et $\widehat{C_1}$ sont correspondants et
        $AB\sslash DC$ (propriétés d'un parallélogramme), donc

\begin{exercise}
    \begin{wrapfigure}{r}{.3\linewidth}
        \includegraphics[width=\linewidth]{images/cocktail.jpg}
    \end{wrapfigure}
    Un délicieux jus de fruits frais est composé de :
    \begin{itemize}
        \item $\dfrac{2}{3}$ de jus d'oranges.
        \item $\dfrac{1}{3}$ de jus de pamplemousses.
    \end{itemize}

    Il faut presser \SI{1}{kg} d'oranges pour obtenir \SI{40}{cl} de jus et
    \SI{1}{kg} de pamplemousses pour obtenir \SI{25}{cl} de jus.\par
    Dans le commerce tu peux trouver des oranges et des pamplemousses au prix
    suivant :
    \begin{description}
        \item[Oranges] 12€/\SI{5}{kg}
        \item[Pamplemousses] 10,80€/\SI{4}{kg}
    \end{description}
    Que coûte la préparation de \SI{4,5}{l} de ce cocktail de fruis frais si
    tu achètes la quantité exacte de fruis ?
\end{exercise}
\begin{solution}
    Si je veux préparer \SI{4,5}{l} de cocktail dans les proportions données,
    il me faut,
    \begin{align*}
         & \text{Jus d'orange}                               &  & \text{Jus de pamplemousse}                         \\
         & \dfrac{2}{3}\cdot 4,5  = \SI{3}{l} = \SI{300}{cl} &  & \dfrac{1}{3}\cdot 4,5 = \SI{1,5}{l} = \SI{150}{cl} \\
    \end{align*}
    Déterminons maintenant le nombre de kilogrammes à acheter pour chaque fruit,
    \begin{align*}
         & \text{Orange}                                                                        &  & \text{Pamplemousse}                                                                \\
         & \SI{40}{cl}\longrightarrow\SI{1}{kg}                                                 &  & \SI{25}{cl}\longrightarrow\SI{1}{kg}                                               \\
         & \SI{1}{cl}\longrightarrow\SI[parse-numbers=false]{\frac{1}{40}}{kg}                  &  & \SI{1}{cl}\longrightarrow\SI[parse-numbers=false]{\frac{1}{25}}{kg}                \\
         & \SI{300}{cl}\longrightarrow\SI[parse-numbers=false]{\frac{300}{40}}{kg}=\SI{7,5}{kg} &  & \SI{150}{cl}\longrightarrow\SI[parse-numbers=false]{\frac{150}{25}}{kg}=\SI{6}{kg} \\
    \end{align*}
    Par la même logique, nous pouvons calculer le prix pour chaque fruit
    \begin{align*}
         & \text{Orange}                         &  & \text{Pamplemousse}                   \\
         & \SI{5}{kg}\longrightarrow\SI{12}{€}   &  & \SI{4}{kg}\longrightarrow\SI{10,8}{€} \\
         & \SI{2,5}{kg}\longrightarrow\SI{6}{€}  &  & \SI{2}{kg}\longrightarrow\SI{5,4}{€}  \\
         & \SI{7,5}{kg}\longrightarrow\SI{18}{€} &  & \SI{6}{kg}\longrightarrow\SI{16,2}{€}
    \end{align*}
    Préparer \SI{4,5}{l} de ce cocktail coûtera donc $18+16,2=\SI{34,2}{€}$.
\end{solution}
