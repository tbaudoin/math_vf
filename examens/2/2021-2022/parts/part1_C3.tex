\begin{exercise}[points=4]
    $ABCDEFGH$ est un octogone régulier. Réponds aux questions suivantes.
    \begin{center}
        \input{tikz/octogone.tikz}
    \end{center}
    \begin{enumerate}
        \item Quelle est l'image du point $G$ par la translation qui applique $A$ sur $C$ ? \blank{$E$}
        \item $R_{O,-90\degres}(C)=?$ \blank{$B$}
        \item Quelle est l'image du triangle $AOB$ par la symétrie orthogonale d'axe $HD$ ? \blank{$GOF$}
        \item $S_{O}([DB])=?$ \blank{$[HF]$}
    \end{enumerate}
\end{exercise}
%SOLUTION
\begin{solution}
    \begin{enumerate}[cols=4]
        \item $E$
        \item $B$
        \item $GOF$
        \item $[HF]$
    \end{enumerate}
\end{solution}
